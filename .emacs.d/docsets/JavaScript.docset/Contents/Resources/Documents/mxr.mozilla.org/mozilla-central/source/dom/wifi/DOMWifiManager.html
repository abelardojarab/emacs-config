144   _convertWifiNetworkToJSON: function(aNetwork) {
145     let json = {};
146 
147     for (let key in aNetwork) {
148       // In WifiWorker.js there are lots of check using "key in network".
149       // So if the value of any property of WifiNetwork is undefined, do not clone it.
150       if (aNetwork[key] != undefined) {
151         json[key] = aNetwork[key];
152       }
153     }
154     return json;
155   },

191   receiveMessage: function(aMessage) {
192     let msg = aMessage.json;
193     if (msg.mid && msg.mid != this._id)
194       return;
195 
196     let request;
197     if (msg.rid) {
198       request = this.takeRequest(msg.rid);
199       if (!request) {
200         return;
201       }
202     }
203 
204     switch (aMessage.name) {
205       case "WifiManager:setWifiEnabled:Return:OK":
206         Services.DOMRequest.fireSuccess(request, msg.data);
207         break;
208 
209       case "WifiManager:setWifiEnabled:Return:NO":
210         Services.DOMRequest.fireError(request, "Unable to enable/disable Wifi");
211         break;
212 
213       case "WifiManager:getNetworks:Return:OK":
214         Services.DOMRequest.fireSuccess(request, this._convertWifiNetworks(msg.data));
215         break;
216 
217       case "WifiManager:getNetworks:Return:NO":
218         Services.DOMRequest.fireError(request, "Unable to scan for networks");
219         break;
220 
221       case "WifiManager:getKnownNetworks:Return:OK":
222         Services.DOMRequest.fireSuccess(request, this._convertWifiNetworks(msg.data));
223         break;
224 
225       case "WifiManager:getKnownNetworks:Return:NO":
226         Services.DOMRequest.fireError(request, "Unable to get known networks");
227         break;
228 
229       case "WifiManager:associate:Return:OK":
230         Services.DOMRequest.fireSuccess(request, true);
231         break;
232 
233       case "WifiManager:associate:Return:NO":
234         Services.DOMRequest.fireError(request, "Unable to add the network");
235         break;
236 
237       case "WifiManager:forget:Return:OK":
238         Services.DOMRequest.fireSuccess(request, true);
239         break;
240 
241       case "WifiManager:forget:Return:NO":
242         Services.DOMRequest.fireError(request, msg.data);
243         break;
244 
245       case "WifiManager:wps:Return:OK":
246         Services.DOMRequest.fireSuccess(request, msg.data);
247         break;
248 
249       case "WifiManager:wps:Return:NO":
250         Services.DOMRequest.fireError(request, msg.data);
251         break;
252 
253       case "WifiManager:setPowerSavingMode:Return:OK":
254         Services.DOMRequest.fireSuccess(request, msg.data);
255         break;
256 
257       case "WifiManager:setPowerSavingMode:Return:NO":
258         Services.DOMRequest.fireError(request, msg.data);
259         break;
260 
261       case "WifiManager:setHttpProxy:Return:OK":
262         Services.DOMRequest.fireSuccess(request, msg.data);
263         break;
264 
265       case "WifiManager:setHttpProxy:Return:NO":
266         Services.DOMRequest.fireError(request, msg.data);
267         break;
268 
269       case "WifiManager:setStaticIpMode:Return:OK":
270         Services.DOMRequest.fireSuccess(request, msg.data);
271         break;
272 
273       case "WifiManager:setStaticIpMode:Return:NO":
274         Services.DOMRequest.fireError(request, msg.data);
275         break;
276 
277       case "WifiManager:importCert:Return:OK":
278         Services.DOMRequest.fireSuccess(request, Cu.cloneInto(msg.data, this._window));
279         break;
280 
281       case "WifiManager:importCert:Return:NO":
282         Services.DOMRequest.fireError(request, msg.data);
283         break;
284 
285       case "WifiManager:getImportedCerts:Return:OK":
286         Services.DOMRequest.fireSuccess(request, Cu.cloneInto(msg.data, this._window));
287         break;
288 
289       case "WifiManager:getImportedCerts:Return:NO":
290         Services.DOMRequest.fireError(request, msg.data);
291         break;
292 
293       case "WifiManager:deleteCert:Return:OK":
294         Services.DOMRequest.fireSuccess(request, msg.data);
295         break;
296 
297       case "WifiManager:deleteCert:Return:NO":
298         Services.DOMRequest.fireError(request, msg.data);
299         break;
300 
301       case "WifiManager:wifiDown":
302         this._enabled = false;
303         this._currentNetwork = null;
304         this._fireEnabledOrDisabled(false);
305         break;
306 
307       case "WifiManager:wifiUp":
308         this._enabled = true;
309         this._macAddress = msg.macAddress;
310         this._fireEnabledOrDisabled(true);
311         break;
312 
313       case "WifiManager:onconnecting":
314         this._currentNetwork = this._convertWifiNetwork(msg.network);
315         this._connectionStatus = "connecting";
316         this._fireStatusChangeEvent(msg.network);
317         break;
318 
319       case "WifiManager:onassociate":
320         this._currentNetwork = this._convertWifiNetwork(msg.network);
321         this._connectionStatus = "associated";
322         this._fireStatusChangeEvent(msg.network);
323         break;
324 
325       case "WifiManager:onconnect":
326         this._currentNetwork = this._convertWifiNetwork(msg.network);
327         this._connectionStatus = "connected";
328         this._fireStatusChangeEvent(msg.network);
329         break;
330 
331       case "WifiManager:ondisconnect":
332         this._currentNetwork = null;
333         this._connectionStatus = "disconnected";
334         this._lastConnectionInfo = null;
335         this._fireStatusChangeEvent(msg.network);
336         break;
337 
338       case "WifiManager:onwpstimeout":
339         this._currentNetwork = null;
340         this._connectionStatus = "wps-timedout";
341         this._lastConnectionInfo = null;
342         this._fireStatusChangeEvent(msg.network);
343         break;
344 
345       case "WifiManager:onwpsfail":
346         this._currentNetwork = null;
347         this._connectionStatus = "wps-failed";
348         this._lastConnectionInfo = null;
349         this._fireStatusChangeEvent(msg.network);
350         break;
351 
352       case "WifiManager:onwpsoverlap":
353         this._currentNetwork = null;
354         this._connectionStatus = "wps-overlapped";
355         this._lastConnectionInfo = null;
356         this._fireStatusChangeEvent(msg.network);
357         break;
358 
359       case "WifiManager:connectioninfoupdate":
360         this._lastConnectionInfo = this._convertConnectionInfo(msg);
361         this._fireConnectionInfoUpdate(msg);
362         break;
363       case "WifiManager:onconnectingfailed":
364         this._currentNetwork = null;
365         this._connectionStatus = "connectingfailed";
366         this._lastConnectionInfo = null;
367         this._fireStatusChangeEvent(msg.network);
368         break;
369       case "WifiManager:onauthenticating":
370         this._currentNetwork = this._convertWifiNetwork(msg.network);
371         this._connectionStatus = "authenticating";
372         this._fireStatusChangeEvent(msg.network);
373         break;
374       case "WifiManager:stationinfoupdate":
375         this._stationNumber = msg.station;
376         this._fireStationInfoUpdate(msg);
377         break;
378     }
379   },
